#ifndef arduinoFFT_h
#define arduinoFFT_h

#include "Arduino.h"

#define FFT_LIB_VERSION "1.5.6"

enum fft_window_t {
	FFT_WIN_TYP_RECTANGLE, /* rectangle (Box car) */
	FFT_WIN_TYP_HAMMING, /* hamming */
	FFT_WIN_TYP_HANN, /* hann */
	FFT_WIN_TYP_TRIANGLE, /* triangle (Bartlett) */
	FFT_WIN_TYP_NUTTALL, /* nuttall */
	FFT_WIN_TYP_BLACKMAN, /* blackman */
	FFT_WIN_TYP_BLACKMAN_NUTTALL, /* blackman nuttall */
	FFT_WIN_TYP_BLACKMAN_HARRIS, /* blackman harris*/
	FFT_WIN_TYP_FLT_TOP, /* flat top */
	FFT_WIN_TYP_WELCH /* welch */
};

enum fft_direction_t {
	FFT_FORWARD, /* Forward FFT */
	FFT_REVERSE /* Reverse FFT */
};

class arduinoFFT {
public:
	arduinoFFT(void);
	arduinoFFT(double *vReal, double *vImag, uint16_t samples, double samplingFrequency);
	void ComplexToMagnitude(void);
	void Compute(fft_direction_t dir);
	void DCRemoval(void);
	double MajorPeak(void);
	void Reorder(void);
	void Windowing(fft_window_t windowType, fft_direction_t dir);
private:
	void Swap(double *x, double *y);
	uint16_t _samples;
	double _samplingFrequency;
	double *_vReal;
	double *_vImag;
};

#endif
