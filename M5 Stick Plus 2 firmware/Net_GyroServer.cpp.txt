const char* GYRO_HTML = R"rawliteral(
<!DOCTYPE html>
<html>
<head>
    <title>M5-Nexus IMU 3D</title>
    <meta name="viewport" content="width=device-width, initial-scale=1">
    <style>
        body { margin: 0; background-color: #1e1e1e; color: #d4d4d4; overflow: hidden; font-family: monospace; }
        canvas { display: block; }
        #info { position: absolute; top: 10px; width: 100%; text-align: center; z-index: 100; }
    </style>
    <script type="importmap">
    {
        "imports": {
            "three": "https://cdn.jsdelivr.net/npm/three@0.166.1/build/three.module.js",
            "three/examples/jsm/": "https://cdn.jsdelivr.net/npm/three@0.166.1/examples/jsm/"
        }
    }
    </script>
</head>
<body>
    <div id="info">M5-Nexus IMU Visualization</div>

    <script type="module">
        import * as THREE from 'three';
        import { OrbitControls } from 'three/examples/jsm/controls/OrbitControls.js';
        import Stats from 'three/examples/jsm/libs/stats.module.js';

        let scene, camera, renderer, box, controls, stats;
        let isUserControlling = false;

        // --- Sensor Fusion and Physics State ---
        let deviceQuaternion = new THREE.Quaternion();
        let devicePosition = new THREE.Vector3();
        let deviceVelocity = new THREE.Vector3();
        let lastTimestamp = performance.now();
        const dps_to_rad = Math.PI / 180.0;


        function init() {
            // Scene
            scene = new THREE.Scene();

            // Camera
            camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
            camera.position.z = 5;

            // Renderer
            renderer = new THREE.WebGLRenderer({ antialias: true });
            renderer.setSize(window.innerWidth, window.innerHeight);
            document.body.appendChild(renderer.domElement);
            
            // Stats (FPS counter)
            stats = new Stats();
            document.body.appendChild(stats.dom);

            // Controls
            controls = new OrbitControls(camera, renderer.domElement);
            controls.enableDamping = true;
            controls.addEventListener('start', () => { isUserControlling = true; });
            controls.addEventListener('end', () => { isUserControlling = false; });

            // Box (representing M5Stick) - Proportions adjusted to match screen ratio (135x240)
            const geometry = new THREE.BoxGeometry(1.35, 2.4, 0.6); 
            const material = new THREE.MeshStandardMaterial({ color: 0xfc9803, roughness: 0.5 }); // Orange color
            box = new THREE.Mesh(geometry, material);
            scene.add(box);

            // Lights
            const ambientLight = new THREE.AmbientLight(0xffffff, 0.8);
            scene.add(ambientLight);
            const directionalLight = new THREE.DirectionalLight(0xffffff, 1.2);
            directionalLight.position.set(2, 5, 5);
            scene.add(directionalLight);

            // Handle window resize
            window.addEventListener('resize', onWindowResize, false);
            
            // Start fetching data and animating
            fetchData();
            animate();
        }

        function onWindowResize() {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        }

        async function fetchData() {
            try {
                const response = await fetch('/data');
                if (!response.ok) {
                    throw new Error(`HTTP error! status: ${response.status}`);
                }
                const data = await response.json();
                
                if (isUserControlling) {
                    // When user is controlling, reset physics to avoid jumps when releasing
                    lastTimestamp = performance.now();
                    deviceVelocity.set(0, 0, 0);
                } else {
                    const now = performance.now();
                    const dt = Math.min((now - lastTimestamp) / 1000.0, 0.1); // Cap dt to avoid physics explosion
                    lastTimestamp = now;

                    // --- 1. ORIENTATION (Sensor Fusion: Gyro + Accelerometer) ---
                    
                    // Gyroscope update (prediction)
                    const gx = -data.gx * dps_to_rad;
                    const gy = -data.gy * dps_to_rad;
                    const gz =  data.gz * dps_to_rad;
                    const deltaQuat = new THREE.Quaternion().setFromEuler(new THREE.Euler(gy, gx, gz, 'YXZ'));
                    deviceQuaternion.premultiply(deltaQuat);
                    
                    // Accelerometer correction
                    const accelVec = new THREE.Vector3(-data.ax, -data.ay, data.az);
                    if (accelVec.lengthSq() > 0.5) { // Only correct if accel is near 1G
                        const gravity = accelVec.clone().normalize();
                        const modelZ = gravity.clone().negate();
                        const worldUp = new THREE.Vector3(0, 1, 0);
                        const modelX = new THREE.Vector3().crossVectors(worldUp, modelZ).normalize();
                        if (modelX.lengthSq() < 0.001) { // Gimbal lock check
                            modelX.crossVectors(new THREE.Vector3(1, 0, 0), modelZ).normalize();
                        }
                        const modelY = new THREE.Vector3().crossVectors(modelZ, modelX);
                        const rotationMatrix = new THREE.Matrix4().makeBasis(modelX, modelY, modelZ);
                        const accelQuat = new THREE.Quaternion().setFromRotationMatrix(rotationMatrix);
                        deviceQuaternion.slerp(accelQuat, 0.05); // Fuse via slerp
                    }
                    deviceQuaternion.normalize();
                    box.quaternion.copy(deviceQuaternion);

                    // --- 2. POSITION (Double integration of linear acceleration) ---

                    const worldGravity = new THREE.Vector3(0, -1, 0); // Gravity vector in world space
                    const localGravity = worldGravity.clone().applyQuaternion(deviceQuaternion.clone().invert());
                    const linearAccel = accelVec.clone().sub(localGravity);
                    const worldLinearAccel = linearAccel.clone().applyQuaternion(deviceQuaternion);
                    
                    const ACCEL_SCALE = 3.0;
                    const SPRING_STIFFNESS = 25.0;
                    const DAMPING = 5.0;

                    let force = worldLinearAccel.multiplyScalar(ACCEL_SCALE);
                    force.add(devicePosition.clone().multiplyScalar(-SPRING_STIFFNESS)); // Spring force
                    force.add(deviceVelocity.clone().multiplyScalar(-DAMPING));       // Damping force
                    
                    deviceVelocity.add(force.clone().multiplyScalar(dt));
                    devicePosition.add(deviceVelocity.clone().multiplyScalar(dt));

                    const MAX_POS = 1.0;
                    if (devicePosition.length() > MAX_POS) {
                         devicePosition.normalize().multiplyScalar(MAX_POS);
                         deviceVelocity.set(0,0,0);
                    }
                    box.position.copy(devicePosition);
                }
            } catch (error) {
                console.error('Error fetching IMU data:', error);
                document.getElementById('info').innerText = "Connection Error - Check Console";
            }
            setTimeout(fetchData, 30);
        }

        function animate() {
            requestAnimationFrame(animate);
            controls.update(); // for damping
            stats.update();
            renderer.render(scene, camera);
        }

        init();
    </script>
</body>
</html>
)rawliteral";

void handleGyroRoot() {
    server.send(200, "text/html", GYRO_HTML);
}

void handleGyroData() {
    // Данные с IMU уже обновляются в главном цикле (через M5.update),
    // поэтому мы просто получаем последние доступные значения.
    // Повторный вызов M5.Imu.update() здесь приводил к конфликтам и ошибке 503.
    auto data = M5.Imu.getImuData();
    
    // Отправляем "сырые" данные акселерометра. Вся логика поворота теперь на стороне клиента (JS).
    float ax = data.accel.x;
    float ay = data.accel.y;
    float az = data.accel.z;
    float gx = data.gyro.x;
    float gy = data.gyro.y;
    float gz = data.gyro.z;

    char jsonBuffer[200];
    snprintf(jsonBuffer, sizeof(jsonBuffer), 
             "{\"ax\":%.2f,\"ay\":%.2f,\"az\":%.2f,\"gx\":%.2f,\"gy\":%.2f,\"gz\":%.2f}",
             ax, ay, az, gx, gy, gz);
    
    server.send(200, "application/json", jsonBuffer);
}

void startGyroServer() {
    if (isWifiConnected()) {
        server.on("/", HTTP_GET, handleGyroRoot);
        server.on("/data", HTTP_GET, handleGyroData);
        server.begin();
        Serial.println("Gyroscope web server started.");
    } else {
        Serial.println("Cannot start gyro server, WiFi not connected.");
    }
}

void stopGyroServer() {
    server.stop();
    Serial.println("Gyroscope web server stopped.");
}

void handleGyroServerClient() {
    server.handleClient();
