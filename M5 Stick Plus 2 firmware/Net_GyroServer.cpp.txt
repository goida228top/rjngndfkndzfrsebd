// --- НОВЫЕ ФУНКЦИИ ДЛЯ ВЕБ-СЕРВЕРА ГИРОСКОПА ---

const char* GYRO_HTML = R"rawliteral(
<!DOCTYPE html>
<html>
<head>
    <title>M5-Nexus IMU 3D</title>
    <meta name="viewport" content="width=device-width, initial-scale=1">
    <style>
        body { margin: 0; background-color: #1e1e1e; color: #d4d4d4; overflow: hidden; font-family: monospace; }
        canvas { display: block; }
        #info { position: absolute; top: 10px; width: 100%; text-align: center; z-index: 100; }
    </style>
</head>
<body>
    <div id="info">M5-Nexus IMU Visualization</div>

    <script type="module">
        import * as THREE from 'https://cdn.jsdelivr.net/npm/three@0.166.1/build/three.module.js';

        let scene, camera, renderer, box;

        function init() {
            // Scene
            scene = new THREE.Scene();

            // Camera
            camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
            camera.position.z = 5;

            // Renderer
            renderer = new THREE.WebGLRenderer({ antialias: true });
            renderer.setSize(window.innerWidth, window.innerHeight);
            document.body.appendChild(renderer.domElement);

            // Box (representing M5Stick) - Proportions adjusted to match screen ratio (135x240)
            const geometry = new THREE.BoxGeometry(1.35, 2.4, 0.6); 
            const material = new THREE.MeshStandardMaterial({ color: 0xfc9803, roughness: 0.5 }); // Orange color
            box = new THREE.Mesh(geometry, material);
            scene.add(box);

            // Lights
            const ambientLight = new THREE.AmbientLight(0xffffff, 0.8);
            scene.add(ambientLight);
            const directionalLight = new THREE.DirectionalLight(0xffffff, 1.2);
            directionalLight.position.set(2, 5, 5);
            scene.add(directionalLight);

            // Handle window resize
            window.addEventListener('resize', onWindowResize, false);
            
            // Start fetching data and animating
            fetchData();
            animate();
        }

        function onWindowResize() {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        }

        async function fetchData() {
            try {
                const response = await fetch('/data');
                if (!response.ok) {
                    throw new Error(`HTTP error! status: ${response.status}`);
                }
                const data = await response.json();
                
                // --- GRAVITY VECTOR ORIENTATION LOGIC ---
                // This approach uses the accelerometer's gravity vector to directly orient the model,
                // avoiding gimbal lock and providing intuitive pitch and roll.

                // The reference vector for the back of the M5Stick model in its default orientation.
                const referenceBack = new THREE.Vector3(0, 0, -1);

                // The measured gravity vector from the IMU.
                // The IMU values are in G's, so the vector is already close to unit length.
                const gravity = new THREE.Vector3(data.ax, data.ay, data.az);
                
                // Create a quaternion that represents the rotation from the reference vector to the gravity vector.
                const targetQuaternion = new THREE.Quaternion().setFromUnitVectors(referenceBack, gravity);
                
                // Slerp (spherical linear interpolation) to the target for smooth animation.
                box.quaternion.slerp(targetQuaternion, 0.3);

            } catch (error) {
                console.error('Error fetching IMU data:', error);
                document.getElementById('info').innerText = "Connection Error - Check Console";
            }
            // Fetch data frequently for a smoother, more responsive animation.
            setTimeout(fetchData, 30); // ~33 times per second
        }

        function animate() {
            requestAnimationFrame(animate);
            renderer.render(scene, camera);
        }

        init();
    </script>
</body>
</html>
)rawliteral";

void handleGyroRoot() {
    server.send(200, "text/html", GYRO_HTML);
}

void handleGyroData() {
    // Данные с IMU уже обновляются в главном цикле (через M5.update),
    // поэтому мы просто получаем последние доступные значения.
    // Повторный вызов M5.Imu.update() здесь приводил к конфликтам и ошибке 503.
    auto data = M5.Imu.getImuData();
    
    // Отправляем "сырые" данные акселерометра. Вся логика поворота теперь на стороне клиента (JS).
    float ax = data.accel.x;
    float ay = data.accel.y;
    float az = data.accel.z;
    float gx = data.gyro.x;
    float gy = data.gyro.y;
    float gz = data.gyro.z;

    char jsonBuffer[200];
    snprintf(jsonBuffer, sizeof(jsonBuffer), 
             "{\"ax\":%.2f,\"ay\":%.2f,\"az\":%.2f,\"gx\":%.2f,\"gy\":%.2f,\"gz\":%.2f}",
             ax, ay, az, gx, gy, gz);
    
    server.send(200, "application/json", jsonBuffer);
}

void startGyroServer() {
    if (isWifiConnected()) {
        server.on("/", HTTP_GET, handleGyroRoot);
        server.on("/data", HTTP_GET, handleGyroData);
        server.begin();
        Serial.println("Gyroscope web server started.");
    } else {
        Serial.println("Cannot start gyro server, WiFi not connected.");
    }
}

void stopGyroServer() {
    server.stop();
    Serial.println("Gyroscope web server stopped.");
}

void handleGyroServerClient() {
    server.handleClient();
}
